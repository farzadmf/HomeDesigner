207  void RotateToPointOfFocus(glm::vec3 pointOfFocus)
208  {
209  auto v1 = normalize(pointOfFocus - Position); // Vector from camera position to focus point
210  auto v2 = normalize(ViewDirection - Position); // Vector from camera position along its view direction
211  auto dotProduct = dot(v1, v2); // Angle between the two vectors
212  auto angle = glm::degrees(acos(dotProduct)); // Angle in degrees
