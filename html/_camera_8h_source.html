213  void RotateToPointOfFocus(glm::vec3 pointOfFocus)
214  {
215  auto v1 = normalize(pointOfFocus - Position); // Vector from camera position to focus point
216  auto v2 = normalize(ViewDirection - Position); // Vector from camera position along its view direction
217  auto dotProduct = dot(v1, v2); // Angle between the two vectors
218  auto angle = glm::degrees(acos(dotProduct)); // Angle in degrees
219  auto crossProduct = cross(v1, v2); // If crossProduct.y > 0, focus is on the right; left otherwise
220 
221  ViewDirection = pointOfFocus;
222  Up = glm::vec3(0.0f, 1.0f, 0.0f);
223  Yaw += crossProduct.y > 0 ? angle : -angle;
224  UpdateCameraVectors();
225  }
